\section{appmate}

\begin{frame}{appmate overview}
  \Large

  \begin{itemize}
    \item 簡單的 Web application

    \item Django project

    \item 所以是使用 Python

    \item 提供 與 database 互動的 http API

    \item API 採用 REST design

    \item 我們把它設計成 template project
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{appmate setup}

  建立 database, 這裡預設是使用 SQLite
  \begin{block}{code}
  \begin{verbatim}
    python ./maange.py migrate
  \end{verbatim}
  \end{block}

  建立 Django 後臺的 admin user
  \begin{block}{code}
  \begin{verbatim}
    python ./manage.py createsuperuser
  \end{verbatim}
  \end{block}
\end{frame}

\begin{frame}[fragile]{appmate setup}
  Run the server!
  \begin{block}{code}
  \begin{verbatim}
    python ./manage.py runserver
  \end{verbatim}
  \end{block}

  並且仔細看看畫面上的訊息
\end{frame}

  \end{itemize}
\end{frame}
